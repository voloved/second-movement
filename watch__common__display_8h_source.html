26// Union representing a single segment mapping
27// COM occupies two bits, SEG occupes the rest.
28typedef union segment_mapping_t {
29 struct {
30 uint8_t com : 2;
31 uint8_t seg : 6;
32 } address;
33 uint8_t value;
34} segment_mapping_t;
35
36// Value to indicate that a segment does not exist
37static const uint8_t segment_does_not_exist = 0xff;
38
39// Union representing 8 segment mappings, A-H
40typedef union digit_mapping_t {
41 segment_mapping_t segment[8];
42 uint64_t value;
43} digit_mapping_t;
